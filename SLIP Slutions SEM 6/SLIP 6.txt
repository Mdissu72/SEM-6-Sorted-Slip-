import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class BlinkImage extends JFrame implements ActionListener {
    private JLabel imageLabel;
    private Timer timer;
    private boolean isVisible = true;

    public BlinkImage() {
        setTitle("Blinking Image");
        setSize(500, 500);
        setLayout(new BorderLayout());
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Load and set the image
        ImageIcon icon = new ImageIcon("image.jpg"); // Add your image path here
        imageLabel = new JLabel(icon);
        add(imageLabel, BorderLayout.CENTER);

        // Timer to toggle visibility every 500ms
        timer = new Timer(500, this);
        timer.start();
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        isVisible = !isVisible;
        imageLabel.setVisible(isVisible);
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> {
            BlinkImage frame = new BlinkImage();
            frame.setVisible(true);
        });
    }
}

VisitServlet.java

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class VisitServlet extends HttpServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        
        int visitCount = 0;
        boolean isNewVisitor = true;

        // Get cookies
        Cookie[] cookies = request.getCookies();
        if (cookies != null) {
            for (Cookie cookie : cookies) {
                if (cookie.getName().equals("visitCount")) {
                    visitCount = Integer.parseInt(cookie.getValue());
                    isNewVisitor = false;
                }
            }
        }

        if (isNewVisitor) {
            out.println("<h3>Welcome! This is your first visit.</h3>");
        } else {
            visitCount++;
            out.println("<h3>Welcome back! You have visited this page " + visitCount + " times.</h3>");
        }

        // Set cookie for visit count
        Cookie visitCookie = new Cookie("visitCount", String.valueOf(visitCount + 1));
        visitCookie.setMaxAge(24 * 60 * 60); // Cookie valid for 1 day
        response.addCookie(visitCookie);
    }
}

package com.example.arithmeticapp;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText num1, num2;
    TextView result;
    Button addBtn, subBtn, mulBtn, divBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        num1 = findViewById(R.id.num1);
        num2 = findViewById(R.id.num2);
        result = findViewById(R.id.result);
        addBtn = findViewById(R.id.addBtn);
        subBtn = findViewById(R.id.subBtn);
        mulBtn = findViewById(R.id.mulBtn);
        divBtn = findViewById(R.id.divBtn);

        addBtn.setOnClickListener(v -> calculate('+'));
        subBtn.setOnClickListener(v -> calculate('-'));
        mulBtn.setOnClickListener(v -> calculate('*'));
        divBtn.setOnClickListener(v -> calculate('/'));
    }

    private void calculate(char operator) {
        double n1 = Double.parseDouble(num1.getText().toString());
        double n2 = Double.parseDouble(num2.getText().toString());
        double res = 0;

        switch (operator) {
            case '+':
                res = n1 + n2;
                break;
            case '-':
                res = n1 - n2;
                break;
            case '*':
                res = n1 * n2;
                break;
            case '/':
                if (n2 != 0) {
                    res = n1 / n2;
                } else {
                    result.setText("Cannot divide by zero!");
                    return;
                }
                break;
        }
        result.setText("Result: " + res);
    }
}

MainActivity.java

package com.example.notificationapp;

import android.app.NotificationChannel;
import android.app.NotificationManager;
import android.content.Intent;
import android.os.Build;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.app.NotificationCompat;
import androidx.core.app.NotificationManagerCompat;

public class MainActivity extends AppCompatActivity {
    private final String CHANNEL_ID = "my_channel";
    private final int NOTIFICATION_ID = 101;
    Button notifyBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        notifyBtn = findViewById(R.id.notifyBtn);

        // Create Notification Channel
        createNotificationChannel();

        notifyBtn.setOnClickListener(v -> {
            sendNotification();
            // Open second activity after notification click
            Intent intent = new Intent(MainActivity.this, SecondActivity.class);
            startActivity(intent);
        });
    }

    private void createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            CharSequence name = "My Channel";
            String description = "Notification Channel for Test";
            int importance = NotificationManager.IMPORTANCE_DEFAULT;
            NotificationChannel channel = new NotificationChannel(CHANNEL_ID, name, importance);
            channel.setDescription(description);

            NotificationManager manager = getSystemService(NotificationManager.class);
            manager.createNotificationChannel(channel);
        }
    }

    private void sendNotification() {
        NotificationCompat.Builder builder = new NotificationCompat.Builder(this, CHANNEL_ID)
                .setSmallIcon(R.drawable.ic_launcher_foreground)
                .setContentTitle("Notification Sent")
                .setContentText("Click to view the message!")
                .setPriority(NotificationCompat.PRIORITY_DEFAULT);

        NotificationManagerCompat notificationManager = NotificationManagerCompat.from(this);
        notificationManager.notify(NOTIFICATION_ID, builder.build());
    }
}
