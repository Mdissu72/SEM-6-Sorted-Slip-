import java.util.ArrayList;
import java.util.List;

public class EmployeeFilter {
    public static void main(String[] args) {
        // List of employee names
        List<String> employees = new ArrayList<>();
        employees.add("Amit");
        employees.add("Rahul");
        employees.add("Anjali");
        employees.add("Pooja");
        employees.add("Arjun");
        employees.add("Suresh");

        // Display employees whose names start with 'A'
        System.out.println("Employees whose names start with 'A':");
        for (String name : employees) {
            if (name.startsWith("A")) {
                System.out.println(name);
            }
        }
    }
}

âœ… B) Java Program in Multithreading Using Applet for Digital Watch

DigitalWatch.java

import java.applet.Applet;
import java.awt.*;
import java.text.SimpleDateFormat;
import java.util.Date;

public class DigitalWatch extends Applet implements Runnable {
    private Thread t;
    private String time;

    public void init() {
        setSize(300, 150);
        t = new Thread(this);
        t.start();
    }

    public void paint(Graphics g) {
        g.setFont(new Font("Arial", Font.BOLD, 30));
        g.setColor(Color.BLUE);
        g.drawString(time, 50, 80);
    }

    public void run() {
        try {
            while (true) {
                Date now = new Date();
                SimpleDateFormat formatter = new SimpleDateFormat("hh:mm:ss a");
                time = formatter.format(now);
                repaint();
                Thread.sleep(1000);
            }
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

MainActivity.java

package com.example.phonecallapp;

import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    Button callButton;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        callButton = findViewById(R.id.callButton);

        callButton.setOnClickListener(v -> {
            Intent intent = new Intent(Intent.ACTION_DIAL);
            intent.setData(Uri.parse("tel:+919876543210"));
            startActivity(intent);
        });
    }
}

MainActivity.java

package com.example.frienddialogapp;

import android.app.AlertDialog;
import android.content.DialogInterface;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    Button showDialogButton;
    String[] friends = {"Amit", "Rahul", "Neha", "Pooja"};

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        showDialogButton = findViewById(R.id.showDialogButton);

        showDialogButton.setOnClickListener(v -> showFriendDialog());
    }

    private void showFriendDialog() {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setTitle("Choose a Friend");

        builder.setItems(friends, (dialog, which) -> {
            String selectedFriend = friends[which];
            greetFriend(selectedFriend);
        });

        builder.setNegativeButton("Cancel", (dialog, which) -> dialog.dismiss());
        builder.show();
    }

    private void greetFriend(String name) {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setTitle("Greeting");
        builder.setMessage("Hello, " + name + "! How are you?");
        builder.setPositiveButton("OK", (dialog, which) -> dialog.dismiss());
        builder.show();
    }
}
