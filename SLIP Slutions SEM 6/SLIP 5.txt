âœ… B) Java Program in Multithreading Using Applet for Traffic Signal
TrafficSignalApplet.java

import java.applet.*;
import java.awt.*;

public class TrafficSignalApplet extends Applet implements Runnable {
    private String currentSignal = "RED";
    private Thread t;

    public void init() {
        t = new Thread(this);
        t.start();
    }

    public void paint(Graphics g) {
        g.setColor(Color.BLACK);
        g.drawRect(50, 50, 100, 300);
        g.drawRect(60, 60, 80, 80);
        g.drawRect(60, 150, 80, 80);
        g.drawRect(60, 240, 80, 80);

        if (currentSignal.equals("RED")) {
            g.setColor(Color.RED);
            g.fillOval(70, 70, 60, 60);
        } else {
            g.setColor(Color.BLACK);
            g.fillOval(70, 70, 60, 60);
        }

        if (currentSignal.equals("YELLOW")) {
            g.setColor(Color.YELLOW);
            g.fillOval(70, 160, 60, 60);
        } else {
            g.setColor(Color.BLACK);
            g.fillOval(70, 160, 60, 60);
        }

        if (currentSignal.equals("GREEN")) {
            g.setColor(Color.GREEN);
            g.fillOval(70, 250, 60, 60);
        } else {
            g.setColor(Color.BLACK);
            g.fillOval(70, 250, 60, 60);
        }
    }

    public void run() {
        while (true) {
            try {
                currentSignal = "RED";
                repaint();
                Thread.sleep(3000);

                currentSignal = "YELLOW";
                repaint();
                Thread.sleep(2000);

                currentSignal = "GREEN";
                repaint();
                Thread.sleep(3000);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
    }
}

package com.example.textstyleapp;

import android.graphics.Color;
import android.graphics.Typeface;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    TextView collegeName;
    Button changeStyleBtn;
    boolean isChanged = false;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        collegeName = findViewById(R.id.collegeName);
        changeStyleBtn = findViewById(R.id.changeStyleBtn);

        changeStyleBtn.setOnClickListener(v -> {
            if (!isChanged) {
                collegeName.setTextColor(Color.RED);
                collegeName.setTextSize(24);
                collegeName.setTypeface(Typeface.SERIF, Typeface.BOLD_ITALIC);
                isChanged = true;
            } else {
                collegeName.setTextColor(Color.BLUE);
                collegeName.setTextSize(18);
                collegeName.setTypeface(Typeface.SANS_SERIF, Typeface.NORMAL);
                isChanged = false;
            }
        });
    }
}

âœ… B) Android Application to Find Factorial and Display Result in Alert Box

MainActivity.java

package com.example.factorialapp;

import android.app.AlertDialog;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText numberInput;
    Button calculateBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        numberInput = findViewById(R.id.numberInput);
        calculateBtn = findViewById(R.id.calculateBtn);

        calculateBtn.setOnClickListener(v -> {
            String numStr = numberInput.getText().toString();
            if (!numStr.isEmpty()) {
                int num = Integer.parseInt(numStr);
                long factorial = calculateFactorial(num);
                showResult(num, factorial);
            }
        });
    }

    private long calculateFactorial(int n) {
        long fact = 1;
        for (int i = 1; i <= n; i++) {
            fact *= i;
        }
        return fact;
    }

    private void showResult(int number, long result) {
        AlertDialog.Builder builder = new AlertDialog.Builder(this);
        builder.setTitle("Factorial Result");
        builder.setMessage("Factorial of " + number + " is: " + result);
        builder.setPositiveButton("OK", (dialog, which) -> dialog.dismiss());
        builder.show();
    }
}
